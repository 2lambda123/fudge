    else :
        reaction = '%s%d__%s%d_' % ( endl_Z.endl_ZSymbol( ZAZ ), ZAA, endl_Z.endl_ZSymbol( yiZ ), yiA )

    outGoing = []
    print yos
    for yo in yos :

            outGoing.append( [ yo ] )
    outGoing.append( [ residualZA ] )

    for i in outGoing :

    s = ''
